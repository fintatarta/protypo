procedure join2(x; s) is
begin
   for k in x.range loop
     if k /= x.first then
        [#s#]
     end if;

     [#x(k)#]
   end loop;
end join2;

function join(x; s) is
begin
   result := "";
   
   for k in x.range loop
     if k /= x.first then
        result := result + s;
     end if;

     result := result + image(x(k));
   end loop;

   return result;
end join;
