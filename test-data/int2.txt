function average(data) is
begin
  result := 0.0;
  counter := 0;

  for item in data loop
    result := result + item;
    counter := counter+1;
  end loop;

  return result / counter;
end average;

function divmod(x; n) is
begin
  div := x / n;
  mod := x - n*div;

  return div, mod;
end divmod;

function fact(x) is
begin
  if x < 1 then -- bingo
     return 1;
  else
    return x*fact(x-1);
  end if; 
end fact;
